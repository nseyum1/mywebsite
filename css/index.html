  • Monte Carlo Integration: Implemented numerical integration using random sampling to estimate integrals of complex functions. 
    
import numpy as np
import math

def func_a(x):
    if x == 0:
        return 0
    return math.sin(1.0 / x)

def func_b(x):
    return math.exp(-x ** 2)

def monte_carlo(func, lower, upper, n):
    samples = np.random.uniform(lower, upper, n)
    result = np.mean([func(x) for x in samples])
    return (upper - lower) * result

if __name__ == "__main__":
    n = 1000000  # Number of samples

    # Part (a): ∫_0^1 sin(1/x) dx
    result_a = monte_carlo(func_a, 0.0001, 1.0, n)  # Avoid 0 due to singularity
    print(f"A. Estimated value of ∫_0^1 sin(1/x) dx: {result_a:.6f}")

    # Part (b): ∫_{-2}^2 e^(-x^2) dx
    result_b = monte_carlo(func_b, -2.0, 2.0, n)
    print(f"B. Estimated value of ∫_{{-2}}^2 e^(-x^2) dx: {result_b:.6f}")
  • Virus Spread Simulation: Modeled the spread of a virus through a network of computers using probabilistic infection and removal processes. 
    
import numpy as np
import random

def simulate_virus_spread(num_computers=20, prob_spread=0.1, max_infected_removed=5, num_simulations=10000):
    total_days = 0
    total_computers_infected = 0
    times_computer_infected = [0] * num_computers

    for _ in range(num_simulations):
        infected = [0] * num_computers
        ever_infected = [0] * num_computers
        infected[0] = 1
        ever_infected[0] = 1
        day_count = 0
        num_infected_computers = 1

        while num_infected_computers > 0:
            day_count += 1
            new_infections = [0] * num_computers

            for i in range(num_computers):
                if infected[i]:
                    for j in range(num_computers):
                        if not infected[j] and random.random() < prob_spread:
                            new_infections[j] = 1
                            ever_infected[j] = 1

            for i in range(num_computers):
                if new_infections[i]:
                    infected[i] = 1

            indices_infected = [i for i in range(num_computers) if infected[i]]
            num_to_remove = min(len(indices_infected), max_infected_removed)
            for _ in range(num_to_remove):
                idx = random.choice(indices_infected)
                infected[idx] = 0
                indices_infected.remove(idx)

            num_infected_computers = sum(infected)

        total_days += day_count
        total_computers_infected += sum(ever_infected)
        for i in range(num_computers):
            if ever_infected[i]:
                times_computer_infected[i] += 1

    expected_time = total_days / num_simulations
    expected_num_computers_infected = total_computers_infected / num_simulations

    print(f"(a) Expected time to remove the virus from the whole network: {expected_time:.4f} days")
    print("(b) Probability that each computer gets infected at least once:")
    for i in range(num_computers):
        prob = times_computer_infected[i] / num_simulations
        print(f"Computer {i + 1}: {prob:.4f}")
    print(f"(c) Expected number of computers that get infected: {expected_num_computers_infected:.4f}")

if __name__ == "__main__":
    simulate_virus_spread()

char strategy_type(char c) {
    c = toupper((unsigned char)c);
    if (c >= 'A' && c <= 'D') return 'A';
    if (c >= 'E' && c <= 'K') return 'E';
    if (c >= 'L' && c <= 'O') return 'L';
    if (c >= 'P' && c <= 'R') return 'P';
    if (c >= 'S' && c <= 'T') return 'S';
    if (c >= 'U' && c <= 'Y') return 'U';
    if (c == 'Z') return 'Z';
    return 'A';
}

char decode_opponent_move(char my_move, char score_char) {
    int score = score_char - '0';
    if (my_move == 'C') {
        if (score == 3) return 'C';
        else return 'D';
    } else {
        if (score == 5) return 'C';
        else return 'D';
    }
}

char decide_move(char strategy, int first_round, char last_my_move, char last_opp_move, char opp_history[], int round_count, double p) {
    switch(strategy) {
        case 'A':
            return 'C';
        case 'E':
            return 'D';
        case 'L':
            return (rand()%2 == 0) ? 'C' : 'D';
        case 'P': {
            double r = (double)rand()/(double)RAND_MAX;
            return (r < p) ? 'C' : 'D';
        }
        case 'S':
            if (first_round) return 'C';
            return last_opp_move;
        case 'U':
            if (round_count < 2) {
                return 'C';
            } else {
                int len = round_count;
                if (opp_history[len-1] == 'D' && opp_history[len-2] == 'D')
                    return 'D';
                else
                    return 'C';
            }
        case 'Z':
            return 'C';
        default:
            return 'C';
    }
}

void parentProcess(int rounds, int p1_out[2], int p1_in[2], int p2_out[2], int p2_in[2]) {
    char buf1[2], buf2[2];
    int score1 = 0;
    int score2 = 0;
    write(p1_in[1], "R", 1);
    write(p2_in[1], "R", 1);
    read(p1_out[0], buf1, 1);
    read(p2_out[0], buf2, 1);
    int round;
    for (round = 1; round <= rounds; round++) {
        write(p1_in[1], "P", 1);
        write(p2_in[1], "P", 1);
        read(p1_out[0], buf1, 1);
        read(p2_out[0], buf2, 1);
        char move1 = buf1[0];
        char move2 = buf2[0];
        int rscore1 = 0, rscore2 = 0;
        if (move1 == 'C' && move2 == 'C') {rscore1=3; rscore2=3;}
        else if (move1 == 'C' && move2 == 'D') {rscore1=0; rscore2=5;}
        else if (move1 == 'D' && move2 == 'C') {rscore1=5; rscore2=0;}
        else {rscore1=1; rscore2=1;}
        score1 += rscore1;
        score2 += rscore2;
        char out1, out2;
        if (rscore1 == 0) out1='0'; else if (rscore1 ==1) out1='1'; else if (rscore1==3) out1='3'; else out1='5';
        if (rscore2 == 0) out2='0'; else if (rscore2 ==1) out2='1'; else if (rscore2==3) out2='3'; else out2='5';
        write(p1_in[1], &out1, 1);
        write(p2_in[1], &out2, 1);
        printf("Round %d: player 1 %c, player 2 %c. Player 1 scores %d, player 2 scores %d.\n",
               round, move1, move2, rscore1, rscore2);
    }
    write(p1_in[1], "Q", 1);
    write(p2_in[1], "Q", 1);
    double avg1 = (double)score1 / rounds;
    double avg2 = (double)score2 / rounds;
    printf("Final scores: Player 1 = %d, Player 2 = %d.\n", score1, score2);
    printf("Average score per round: Player 1 = %.2f, Player 2 = %.2f\n", avg1, avg2);
}

void playerProcess(int p_out[2], int p_in[2], char strategy) {
    close(p_out[0]);
    close(p_in[1]);
    srand(time(NULL) ^ getpid());
    char cmd;
    int round_count = 0;
    char opp_history[1000];
    char my_history[1000];
    char last_my_move = 'C';
    char last_opp_move = 'C';
    int first_round = 1;
    double p = 0.7;
    while (read(p_in[0], &cmd, 1) == 1) {
        if (cmd == 'R') {
            write(p_out[1], "Y", 1);
        } else if (cmd == 'P') {
            char my_move = decide_move(strategy, first_round, last_my_move, last_opp_move, opp_history, round_count, p);
            write(p_out[1], &my_move, 1);
            my_history[round_count] = my_move;
            last_my_move = my_move;
            first_round = 0;
        } else if (cmd == '0' || cmd == '1' || cmd == '3' || cmd == '5') {
            char opp_move = decode_opponent_move(last_my_move, cmd);
            opp_history[round_count] = opp_move;
            round_count++;
            last_opp_move = opp_move;
        } else if (cmd == 'Q') {
            break;
        }
    }
    close(p_out[1]);
    close(p_in[0]);
}

int main(int argc, char *argv[]) {
    if (argc == 2 && strcmp(argv[1], "-h") == 0) {
        print_help();
        return 0;
    }
    char firstname[50] = "S";
    char lastname[50] = "N";
    if (argc < 2) {
        print_help();
        return 0;
    }
    char *endptr;
    long val = strtol(argv[1], &endptr, 10);
    if (*endptr != '\0' || val <= 0) {
        fprintf(stderr, "Invalid number of rounds: %s\n", argv[1]);
        print_help();
        return 0;
    }
    int rounds = (int)val;
    if (argc >= 3) {
        strncpy(firstname, argv[2], 49);
        firstname[49]='\0';
    }
    if (argc >= 4) {
        strncpy(lastname, argv[3], 49);
        lastname[49]='\0';
    }
    char p1_strategy = strategy_type(firstname[0]);
    char p2_strategy = strategy_type(lastname[0]);
    int p1_out[2], p1_in[2], p2_out[2], p2_in[2];
    if (pipe(p1_out)==-1||pipe(p1_in)==-1||pipe(p2_out)==-1||pipe(p2_in)==-1) {
        perror("pipe");
        exit(1);
    }
    pid_t pid1 = fork();
    if (pid1<0) {
        perror("fork");
        exit(1);
    }
    if (pid1==0) {
        close(p1_out[0]);
        close(p1_in[1]);
        close(p2_out[0]); close(p2_out[1]);
        close(p2_in[0]); close(p2_in[1]);
        playerProcess(p1_out, p1_in, p1_strategy);
        exit(0);
    }
    pid_t pid2 = fork();
    if (pid2<0) {
        perror("fork");
        exit(1);
    }
    if (pid2==0) {
        close(p2_out[0]);
        close(p2_in[1]);
        close(p1_out[0]); close(p1_out[1]);
        close(p1_in[0]); close(p1_in[1]);
        playerProcess(p2_out, p2_in, p2_strategy);
        exit(0);
    }
    close(p1_out[1]);
    close(p1_in[0]);
    close(p2_out[1]);
    close(p2_in[0]);
    parentProcess(rounds,p1_out,p1_in,p2_out,p2_in);
    wait(NULL);
    wait(NULL);
    return 0;
}
